#include <iostream>
#include <stdio.h>

void isPrime(int);

int main() {
    int n;
    scanf_s("%d", &n);
    isPrime(n);
    return 0;
}

void isPrime(int n)
{
    int count = 0;
    if (n > 1)
    {
        for (int i = 2; i < n; i++)
        {
            if (n % i == 0)
            {
                printf("False");
                break;
            }
            count++;
        }   
        if (count == (n - 2))
        {
            printf("True");
        }
    }
    else
        printf("False");
    
}
