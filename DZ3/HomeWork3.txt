int routes(int x, int y)
{
	if (x == 0 && y == 0)
		return 0;
        
    //obstructions
	else if (x == 3 && y == 3)
		return 0;
	else if (x == 5 && y == 7)
		return 0;
	else if (x == 7 && y == 5)
		return 0;
	else if (x == 2 && y == 2)
		return 0;
	else if (x == 4 && y == 1)
		return 0;
	else if (x == 0 || y == 0)

		return 1;
	else
		return routes(x, y - 1) + routes(x - 1, y);
}

void Func(int n) {
	if (n != 0)
		Func(n / 2);
	else return;

	printf("%d", n % 2);
	return;
}

int exp(int a, int b)
{
	int c = a;
	if (b == 0) return 0;
	if (b == 1) return a;
	else{
		for (int i = 1; i < b; i++){
			c *= a;
		}
	}
	return c;
}

int expRec(int a, int b)
{
	if (b == 0) return 0;
	if (b == 1) return a;
	return a * expRec(a, b - 1);
}

int binpow(int a, int b) {
	if (b == 0)
		return 1;
	int x = (b % 2) ? a : 1;
	int c = binpow(a, b / 2);
	return x * c * c;
}
